<script>

  // LocalStorage keys

  const USERS_KEY = 'loanpaypro_users';

  const MEMBERS_KEY = 'loanpaypro_members';

  const LOANS_KEY = 'loanpaypro_loans';

  const PAYMENTS_KEY = 'loanpaypro_payments';

  const SESSION_KEY = 'loanpaypro_session';

  function loadData(key) { return JSON.parse(localStorage.getItem(key)) || []; }

  function saveData(key, data) { localStorage.setItem(key, JSON.stringify(data)); }

  // Authentication

  function signup() {

    let users = loadData(USERS_KEY);

    const username = document.getElementById('signup-username').value;

    const password = document.getElementById('signup-password').value;

    if (!username || !password) return alert("Please fill all fields");

    if (users.find(u => u.username === username)) return alert("Username already exists");

    users.push({ username, password });

    saveData(USERS_KEY, users);

    alert("Account created! Please login.");

    toggleSignup(false);

  }

  function login() {

    let users = loadData(USERS_KEY);

    const username = document.getElementById('login-username').value;

    const password = document.getElementById('login-password').value;

    const user = users.find(u => u.username === username && u.password === password);

    if (user) {

      localStorage.setItem(SESSION_KEY, JSON.stringify(user));

      document.getElementById('auth-section').classList.add('hidden');

      document.getElementById('signup-section').classList.add('hidden');

      document.getElementById('nav').classList.remove('hidden');

      showSection('dashboard');

    } else {

      alert("Invalid login");

    }

  }

  function logout() { localStorage.removeItem(SESSION_KEY); location.reload(); }

  function toggleSignup(show) {

    document.getElementById('signup-section').classList.toggle('hidden', !show);

    document.getElementById('auth-section').classList.toggle('hidden', show);

  }

  // Navigation

  function showSection(id) {

    ['dashboard','members','loans','payments'].forEach(sec => document.getElementById(sec).classList.add('hidden'));

    document.getElementById(id).classList.remove('hidden');

    if (id==='members') renderMembers();

    if (id==='loans') renderLoans();

    if (id==='payments') renderPayments();

  }

  // Members

  function addMember() {

    let members = loadData(MEMBERS_KEY);

    const id = Date.now().toString();

    const name = document.getElementById('member-name').value;

    const email = document.getElementById('member-email').value;

    const phone = document.getElementById('member-phone').value;

    if (!name || !email || !phone) return alert("Fill all fields");

    members.push({ id, name, email, phone });

    saveData(MEMBERS_KEY, members);

    renderMembers();

  }

  function renderMembers() {

    const members = loadData(MEMBERS_KEY);

    document.getElementById('member-list').innerHTML = members.map(m => `

      <tr>

        <td>${m.id}</td>

        <td>${m.name}</td>

        <td>${m.email}</td>

        <td>${m.phone}</td>

        <td><button onclick="sendReminder('${m.id}')">📩 Send Reminder</button></td>

      </tr>

    `).join('');

  }

  // Loans

  function addLoan() {

    let loans = loadData(LOANS_KEY);

    const id = Date.now().toString();

    const memberId = document.getElementById('loan-member-id').value;

    const amount = document.getElementById('loan-amount').value;

    const dueDate = document.getElementById('loan-due-date').value;

    const freq = document.getElementById('loan-frequency').value;

    if (!memberId || !amount || !dueDate) return alert("Fill all fields");

    loans.push({ id, memberId, amount, freq, dueDate });

    saveData(LOANS_KEY, loans);

    renderLoans();

  }

  function renderLoans() {

    const loans = loadData(LOANS_KEY);

    const members = loadData(MEMBERS_KEY);

    document.getElementById('loan-list').innerHTML = loans.map(l => {

      let member = members.find(m => String(m.id) === String(l.memberId)) || { name: "Unknown" };

      return `

        <tr>

          <td>${l.memberId}</td>

          <td>₱${l.amount}</td>

          <td>${l.freq}</td>

          <td>${l.dueDate}</td>

          <td><button onclick="sendReminder('${l.memberId}', '${l.id}')">📩 Send Reminder</button></td>

        </tr>

      `;

    }).join('');

  }

  // Payments

  function addPayment() {

    let payments = loadData(PAYMENTS_KEY);

    const memberId = document.getElementById('payment-member-id').value;

    const amount = document.getElementById('payment-amount').value;

    const status = document.getElementById('payment-status').value;

    if (!memberId || !amount) return alert("Fill all fields");

    payments.push({ memberId, amount, status });

    saveData(PAYMENTS_KEY, payments);

    renderPayments();

  }

  function renderPayments() {

    const list = loadData(PAYMENTS_KEY);

    document.getElementById('payment-list').innerHTML = list.map(p => `

      <tr><td>${p.memberId}</td><td>₱${p.amount}</td><td>${p.status}</td></tr>

    `).join('');

  }

  // Reminders (frontend -> backend)

  function sendEmailReminder(email, message) {

    fetch("http://localhost:3000/send-email", {

      method: "POST", headers: { "Content-Type": "application/json" },

      body: JSON.stringify({ to: email, subject: "Loan Payment Reminder", message })

    }).then(res => res.json()).then(d => alert("Email: " + d.msg)).catch(err => console.error(err));

  }

  function sendSmsReminder(phone, message) {

    fetch("http://localhost:3000/send-sms", {

      method: "POST", headers: { "Content-Type": "application/json" },

      body: JSON.stringify({ phone, message })

    }).then(res => res.json()).then(d => alert("SMS: " + d.msg)).catch(err => console.error(err));

  }

  function sendReminder(memberId, loanId = null) {

    const members = loadData(MEMBERS_KEY);

    const loans = loadData(LOANS_KEY);

    const member = members.find(m => String(m.id) === String(memberId));

    if (!member) return alert("Member not found!");

    let msg = "";

    if (loanId) {

      const loan = loans.find(l => String(l.id) === String(loanId));

      if (!loan) return alert("Loan not found!");

      msg = `Hello ${member.name}, this is a reminder for your loan payment of ₱${loan.amount}, due on ${loan.dueDate}.`;

    } else {

      const memberLoans = loans.filter(l => String(l.memberId) === String(memberId));

      msg = memberLoans.length ?

        `Hello ${member.name}, you have the following loans:\n` + memberLoans.map(l => `₱${l.amount} due on ${l.dueDate}`).join("\n") :

        `Hello ${member.name}, you currently have no active loans.`;

    }

    if (member.email) sendEmailReminder(member.email, msg);

    if (member.phone) sendSmsReminder(member.phone, msg);

  }
