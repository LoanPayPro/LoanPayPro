const express = require("express");

const bodyParser = require("body-parser");

const nodemailer = require("nodemailer");

const twilio = require("twilio");

const cors = require("cors");

const app = express();

app.use(bodyParser.json());

app.use(cors());

// 🔑 Replace with your credentials

const GMAIL_USER = "yourgmail@gmail.com";

const GMAIL_PASS = "your_app_password"; // use App Password

const TWILIO_SID = "your_twilio_sid";

const TWILIO_AUTH = "your_twilio_auth_token";

const TWILIO_PHONE = "+1234567890"; // Twilio phone number

// Email transporter

const transporter = nodemailer.createTransport({

  service: "gmail",

  auth: { user: GMAIL_USER, pass: GMAIL_PASS }

});

// Twilio client

const client = twilio(TWILIO_SID, TWILIO_AUTH);

// Email endpoint

app.post("/send-email", async (req, res) => {

  try {

    const { to, subject, message } = req.body;

    await transporter.sendMail({ from: GMAIL_USER, to, subject, text: message });

    res.json({ msg: "Email sent successfully" });

  } catch (err) {

    console.error(err);

    res.status(500).json({ msg: "Failed to send email" });

  }

});

// SMS endpoint

app.post("/send-sms", async (req, res) => {

  try {

    const { phone, message } = req.body;

    await client.messages.create({ body: message, from: TWILIO_PHONE, to: phone });

    res.json({ msg: "SMS sent successfully" });

  } catch (err) {

    console.error(err);

    res.status(500).json({ msg: "Failed to send SMS" });

  }

});

// Start server

app.listen(3000, () => console.log("Server running on http://localhost:3000"));
